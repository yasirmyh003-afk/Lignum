function isBlank(value) {
  return value === null || value === undefined || String(value).trim() === "";
}

async function loadTruckReleaseData() {
  const { data: inventoryData, error: inventoryError } = await supabaseClient
    .from("inventory")
    .select("*")
    .order("so_number", { ascending: true });

  if (inventoryError) {
    showAppModal({ type:"error", title:"Load Failed", message:"Could not load inventory rows." });
    return;
  }

  inventoryRows = inventoryData || [];

  const { data: carrierData } = await supabaseClient.from("carriers").select("*").order("reference", { ascending:true });
  const { data: originData } = await supabaseClient.from("origins").select("*");
  const { data: customerData } = await supabaseClient.from("customers").select("*");

  carrierRows = carrierData || [];
  originRows = originData || [];
  customerRows = customerData || [];

  fillSOOptions();
  fillCarrierOptions();
}

function fillSOOptions() {
  const select = document.getElementById("so_number");
  const soNumbers = [...new Set(inventoryRows.map(r => r.so_number).filter(v => !isBlank(v)).map(String))].sort();

  select.innerHTML = `<option value="">Select SO#</option>`;
  soNumbers.forEach(so => {
    const option = document.createElement("option");
    option.value = so;
    option.textContent = so;
    select.appendChild(option);
  });
}

function fillCarrierOptions() {
  const select = document.getElementById("carrier_reference");
  select.innerHTML = `<option value="">Select Carrier</option>`;

  carrierRows.forEach(row => {
    const option = document.createElement("option");
    option.value = row.reference || "";
    option.textContent = row.reference || "";
    select.appendChild(option);
  });
}

function handleSOChange() {
  const soNumber = document.getElementById("so_number").value;

  if (!soNumber) {
    clearTruckReleaseForm();
    renderTruckRows([]);
    return;
  }

  const first = inventoryRows.find(row => String(row.so_number) === String(soNumber));

  if (!first) {
    clearTruckReleaseForm();
    renderTruckRows([]);
    return;
  }

  document.getElementById("cust_po_number").value = first.cust_po_number || "";
  document.getElementById("status").value = first.inv || "";
  document.getElementById("customer").value = first.customer || "";
  document.getElementById("location").value = first.location || "";
  document.getElementById("ship_due").value = first.ship_week || "";

  selectedRows = inventoryRows
    .filter(row => String(row.so_number || "") === String(soNumber))
    .sort((a, b) => {
      const carA = String(a.car_number || "");
      const carB = String(b.car_number || "");
      if (carA !== carB) return carA.localeCompare(carB);
      const sizeA = String(a.size || "");
      const sizeB = String(b.size || "");
      if (sizeA !== sizeB) return sizeA.localeCompare(sizeB);
      return Number(a.length_value || 0) - Number(b.length_value || 0);
    });

  renderTruckRows(selectedRows);
}

function renderTruckRows(rows) {
  const tbody = document.getElementById("truck-release-body");
  tbody.innerHTML = "";

  if (!rows.length) {
    tbody.innerHTML = `<tr><td colspan="9" class="tr-empty">Select SO# to load truck release rows.</td></tr>`;
    return;
  }

  let mbfTotal = 0;

  rows.forEach(row => {
    const mbf = Number(row.mbf || 0);
    mbfTotal += mbf;

    const tr = document.createElement("tr");
    tr.innerHTML = `
      <td>${formatNumber(mbf)}</td>
      <td>${row.po_number || ""}</td>
      <td>${row.car_number || ""}</td>
      <td>${row.mill || ""}</td>
      <td>${row.specie || ""}</td>
      <td>${row.grade || ""}</td>
      <td>${row.size || ""}</td>
      <td>${row.length_value || ""}</td>
      <td>${row.ppu || ""}</td>
    `;
    tbody.appendChild(tr);
  });

  const totalRow = document.createElement("tr");
  totalRow.className = "tr-subtotal-row";
  totalRow.innerHTML = `<td>${formatNumber(mbfTotal)}</td><td colspan="8"></td>`;
  tbody.appendChild(totalRow);
}

function clearTruckReleaseForm() {
  ["cust_po_number","status","customer","location","ship_due","additional_email","rate"].forEach(id => {
    document.getElementById(id).value = "";
  });
  selectedRows = [];
}

function formatNumber(value) {
  const num = Number(value || 0);
  if (!num) return "";
  return Number(num.toFixed(3));
}

function formatDateShort() {
  return new Date().toLocaleDateString("en-US", { month:"short", day:"numeric" });
}

function getTemplateType() {
  const inv = String(document.getElementById("status").value || "").toLowerCase();
  if (inv.includes("matt")) return "matt";
  if (inv.includes("quality")) return "quality";
  return "open";
}

function getOriginData() {
  const inv = document.getElementById("status").value;
  const location = document.getElementById("location").value;

  return originRows.find(r =>
    String(r.inv || "").toLowerCase() === String(inv || "").toLowerCase() &&
    String(r.origin || "").toLowerCase() === String(location || "").toLowerCase()
  ) || {};
}

function getDestinationData() {
  const customer = document.getElementById("customer").value;
  return customerRows.find(r => String(r.destination || "") === String(customer || "")) || {};
}

function getCarrierData() {
  const ref = document.getElementById("carrier_reference").value;
  return carrierRows.find(r => String(r.reference || "") === String(ref || "")) || {};
}

async function generateTruckReleasePDF() {
  try {
    if (!document.getElementById("so_number").value || !selectedRows.length) {
      showAppModal({ type:"error", title:"Missing Data", message:"Please select SO# first." });
      return;
    }

    const wrap = document.getElementById("pdf-template-wrap");
    wrap.innerHTML = buildPdfTemplate();

    const canvas = await html2canvas(wrap, { scale: 2, backgroundColor: "#ffffff" });

    const { jsPDF } = window.jspdf;
    const pdf = new jsPDF("landscape", "pt", "a4");
    const pageWidth = pdf.internal.pageSize.getWidth();
    const pageHeight = pdf.internal.pageSize.getHeight();

    pdf.addImage(canvas.toDataURL("image/png"), "PNG", 10, 10, pageWidth - 20, pageHeight - 20);

    const pdfBlob = pdf.output("blob");
    const pdfBase64 = await blobToBase64(pdfBlob);

    const so = document.getElementById("so_number").value;
    const custPO = document.getElementById("cust_po_number").value;
    const customer = document.getElementById("customer").value;
    const inv = document.getElementById("status").value;
    const date = formatDateShort();

    const safeName = `Lignum#${so} Truck Release ${date} ${customer} PO#${custPO}.pdf`.replace(/[\/\\:*?"<>|]/g, "-");

    const { error: uploadError } = await supabaseClient.storage
      .from("truck-release-pdfs")
      .upload(safeName, pdfBlob, { contentType:"application/pdf", upsert:true });

    if (uploadError) throw uploadError;

    const { data: publicData } = supabaseClient.storage
      .from("truck-release-pdfs")
      .getPublicUrl(safeName);

    const pdfUrl = publicData.publicUrl;
    const subject = getTruckReleaseSubject(inv, so, custPO, customer);
    const emails = await getTruckReleaseEmails();

    await saveTruckReleaseLog();

    await supabaseClient.from("truck_release_pdf_logs").insert([{
      pdf_url: pdfUrl,
      so_number: so,
      cust_po_number: custPO,
      customer,
      inv,
      carrier_reference: document.getElementById("carrier_reference").value,
      email_recipients: emails.join(", "),
      email_subject: subject,
      pdf_name: safeName
    }]);

    const emailResponse = await fetch("/api/send-truck-release-email", {
      method:"POST",
      headers:{ "Content-Type":"application/json" },
      body: JSON.stringify({
        toRecipients: emails.map(email => ({ address: email })),
        subject,
        body: `Please find the Truck Release PDF attached.<br><br>Rate: ${document.getElementById("rate").value || ""}`,
        attachmentName: safeName,
        attachmentData: pdfBase64
      })
    });

    if (!emailResponse.ok) {
      showAppModal({ type:"error", title:"Email Failed", message:"PDF saved, but email failed. Check Vercel env variables." });
      window.open(pdfUrl, "_blank");
      return;
    }

    window.open(pdfUrl, "_blank");

    showAppModal({
      type:"success",
      title:"PDF Generated & Sent",
      message:"Truck Release PDF was generated, logged, and emailed."
    });

  } catch (error) {
    console.error("Truck release PDF error:", error);
    showAppModal({ type:"error", title:"PDF Failed", message:error.message || "Could not generate PDF." });
  }
}

async function saveTruckReleaseLog() {
  const header = {
    so_number: document.getElementById("so_number").value,
    cust_po_number: document.getElementById("cust_po_number").value,
    status: document.getElementById("status").value,
    customer: document.getElementById("customer").value,
    carrier_reference: document.getElementById("carrier_reference").value,
    location: document.getElementById("location").value,
    additional_email: document.getElementById("additional_email").value,
    rate: Number(document.getElementById("rate").value || 0),
    ship_due: document.getElementById("ship_due").value
  };

  const logRows = selectedRows.map(row => ({
    ...header,
    mbf: Number(row.mbf || 0),
    ib_number: row.po_number || "",
    car_number: row.car_number || "",
    mill: row.mill || "",
    specie: row.specie || "",
    grade: row.grade || "",
    size: row.size || "",
    length_value: Number(row.length_value || 0),
    pack_size: Number(row.ppu || 0)
  }));

  await supabaseClient.from("truck_release_logs").insert(logRows);
}

function getTruckReleaseSubject(inv, so, custPO, customer) {
  const type = String(inv || "").toLowerCase();
  if (type.includes("matt")) return `Lignum#${so} Truck Release BoL Matt's PO#${custPO}`;
  if (type.includes("quality")) return `Quality PO#${custPO} Truck Release BoL Lignum #${so}`;
  return `Lignum#${so} Truck Release BoL ${customer} PO#${custPO}`;
}

async function getTruckReleaseEmails() {
  const emails = [];
  const customer = document.getElementById("customer").value;
  const carrierRef = document.getElementById("carrier_reference").value;
  const extraEmail = document.getElementById("additional_email").value;

  if (extraEmail) emails.push(extraEmail);

  const cust = customerRows.find(r => String(r.destination || "") === String(customer || ""));
  if (cust) ["email_1","email_2","email_3","email_4"].forEach(k => cust[k] && emails.push(cust[k]));

  const carrier = carrierRows.find(r => String(r.reference || "") === String(carrierRef || ""));
  if (carrier) {
    if (carrier.email_1_fixed) emails.push(carrier.email_1_fixed);
    if (carrier.email_2_fixed) emails.push(carrier.email_2_fixed);
  }

  return [...new Set(emails.join(",").split(/[;, ]+/).map(e => e.trim().toLowerCase()).filter(e => e.includes("@")))];
}

function blobToBase64(blob) {
  return new Promise((resolve, reject) => {
    const reader = new FileReader();
    reader.onloadend = () => resolve(reader.result.split(",")[1]);
    reader.onerror = reject;
    reader.readAsDataURL(blob);
  });
}
